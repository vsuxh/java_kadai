public class Kosensei
{
    private int gakunen;
    private String bunnya;
    private String namae;

    public Kosensei(int year, String major, String name){
    	this.gakunen = year;
    	this.bunnya = major;
    	this.namae = name;
    }

    public int getGakunen() {return this.gakunen;}
    public String getBunnya() {return this.bunnya;}
    public String getNamae (){return this.namae;}
    public void shokai(){
        	System.out.println("私は" + this.gakunen + "年" + this.bunnya + "の" + this.namae + "です.ふぁっきゅー");
    }
}
class KosenseiApp
{
    public static void main(String args[])
    {
    	Kosensei pepsi = new Kosensei(3, "情報", "ペプシメン");
    	pepsi.shokai();
    }
}

public class Ryosei extends Kosensei {
    // フィールド
    private Ryosei doshitsusha;

    // コンストラクタ (学年, 分野, 名前)
    public Ryosei(int year, String major, String name) {
        super(year, major, name); 
        this.doshitsusha = null;  
    }

    public Ryosei getDoshitsusha() {
        return this.doshitsusha;
    }

    public void setDoshitsusha(Ryosei partner) {
        this.doshitsusha = partner;
    }

    public void shokai() {
        System.out.print("私は" + getGakunen() + "年" + getBunnya() + "の" + getNamae() + "です.");
        if (this.doshitsusha != null) {
            System.out.println(" 同室者は " + this.doshitsusha.getNamae() + " です.");
        } else {
            System.out.println(" 同室者はいません.");
        }
    }
}


public class RyoseiApp {
    public static void main(String[] args) {
        Ryosei r1 = new Ryosei(2, "情報", "太郎");
        Ryosei r2 = new Ryosei(3, "機械", "次郎");

        System.out.println("=== 同室者を設定する前 ===");
        r1.shokai();
        r2.shokai();

        r1.setDoshitsusha(r2);
        r2.setDoshitsusha(r1);

        System.out.println("=== 同室者を設定した後 ===");
        r1.shokai();
        r2.shokai();
    }
}
