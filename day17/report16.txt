import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

public class TextAppendGUI extends JFrame implements ActionListener {

    private JTextField tf1;
    private JTextArea ta1;

    public TextAppendGUI() {
        super("入力追加GUI");

        tf1 = new JTextField(20);
        ta1 = new JTextArea(10, 30);

        tf1.addActionListener(this); // Enterキーでイベント発生

        JPanel p = new JPanel();
        p.setLayout(new BorderLayout());
        p.add(tf1, BorderLayout.NORTH);
        p.add(new JScrollPane(ta1), BorderLayout.CENTER);

        add(p);
        setSize(400, 300);
        setDefaultCloseOperation(EXIT_ON_CLOSE);
        setVisible(true);
    }

    public void actionPerformed(ActionEvent e) {
        if (e.getSource() == tf1) {
            String s = tf1.getText();        // 入力取得
            ta1.append(s);                  // テキストエリアへ追加
            ta1.append(System.getProperty("line.separator")); // 改行
            tf1.setText("");                // 入力欄クリア
        }
    }

    public static void main(String[] args) {
        new TextAppendGUI();
    }
}

import javax.swing.*;
import java.awt.*;
import java.awt.event.*;
import java.io.*;

public class FileViewerGUI extends JFrame implements ActionListener {

    private JTextField tfFile;
    private JButton openBtn;
    private JTextArea ta;

    public FileViewerGUI() {
        super("ファイルビューア");

        tfFile = new JTextField(20);
        openBtn = new JButton("開く");
        ta = new JTextArea(15, 40);

        openBtn.addActionListener(this);

        JPanel top = new JPanel();
        top.add(new JLabel("ファイル名:"));
        top.add(tfFile);
        top.add(openBtn);

        add(top, BorderLayout.NORTH);
        add(new JScrollPane(ta), BorderLayout.CENTER);

        setSize(500, 400);
        setDefaultCloseOperation(EXIT_ON_CLOSE);
        setVisible(true);
    }

    public void actionPerformed(ActionEvent e) {
        if (e.getSource() == openBtn) {
            String filename = tfFile.getText();
            String all = "";

            try {
                BufferedReader br = new BufferedReader(new FileReader(filename));
                String buf;
                while ((buf = br.readLine()) != null) {
                    all += buf + System.getProperty("line.separator");
                }
                br.close();
                ta.setText(all);

            } catch (IOException ex) {
                ta.setText("ファイル読込みエラー");
            }
        }
    }

    public static void main(String[] args) {
        new FileViewerGUI();
    }
}
