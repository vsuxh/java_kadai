class Sumaho
{
    private String namae;    // スマホの名前
    private double screen;   // 画面サイズ（インチ）
    private int memory;      // メモリ容量（GB）

    Sumaho(String namae, double screen, int memory){
        this.setNamae(namae);
        this.setScreen(screen);
        this.setMemory(memory);
    }

    public String getNamae(){
        return this.namae;
    }
    public void setNamae(String namae){
        this.namae = namae;
    }

    public double getScreen(){
        return this.screen;
    }
    public void setScreen(double screen){
        if(screen >= 3.0 && screen <= 7.0){
            this.screen = screen;
        } else {
            this.screen = 5.0; // デフォルト値
            System.out.println("画面サイズが範囲外なので 5.0インチに設定しました。");
        }
    }

    public int getMemory(){
        return this.memory;
    }

    public void setMemory(int memory){
        if(memory >= 1 && memory <= 16){
            this.memory = memory;
        } else {
            this.memory = 4; // デフォルト値
            System.out.println("メモリ容量が範囲外なので 4GBに設定しました。");
        }
    }

    public void shokai(){
        System.out.println("スマホ名: " + this.namae + 
                           " / 画面サイズ: " + this.screen + "インチ" +
                           " / メモリ: " + this.memory + "GB");
    }
}

class SumahoApp
{
    public static void main(String args[])
    {
        Sumaho iphone = new Sumaho("iPhone 15", 6.1, 8);
        iphone.shokai();

        Sumaho android = new Sumaho("Galaxy S23", 6.8, 12);
        android.shokai();
        Sumaho cheap = new Sumaho("安スマホ", 8.5, 32);
        cheap.shokai();
        iphone.setScreen(4.5);
        iphone.setMemory(2);
        iphone.shokai();
    }
}

class Kosensei
{
    private int gakunen;
    private String bunnya;
    String namae;
    String jukyo;

    // Sumaho型のフィールド（スマホをまるごと保持）
    private Sumaho sumaho;

    Kosensei(){
        this.gakunen = 0;
        this.bunnya = null;
        this.namae = null;
        this.jukyo = null;
        // オブジェクト変数の初期値は null にする（小文字）
        this.sumaho = null;
    }

    void setGakunen(int grade){
        if ((grade>=1)&&(grade<=5)) this.gakunen = grade;
    }

    void setBunnya(String major){
        if((major=="情報")||(major=="機械")||(major=="電気")||(major=="電子")||(major=="建築")) this.bunnya = major;
    }

    // --------- ここから追加：スマホまるごとのアクセサ ----------
    public void setSumaho(Sumaho s){
        this.sumaho = s;
    }

    public Sumaho getSumaho(){
        return this.sumaho;
    }
    // --------------------------------------------------------

    void shokai(){
        System.out.println("私は" + this.gakunen + "年" + this.bunnya + "の" + this.namae + "です.私の住居は" +  this.jukyo + "です.");
        // スマホを持っているかどうかを判定して表示
        if (this.sumaho != null) {
            System.out.println("私は" + this.sumaho.getNamae() + "を持っています.");
        } else {
            System.out.println("私はスマホを持っていません.");
        }
    }

}

class KosenseiApp
{
    public static void main(String args[])
    {
        Kosensei pepsi = new Kosensei();
        // namae と jukyo はパッケージプライベートなので直接代入（元のコードと同様の扱い）
        pepsi.namae = "Pepsi";
        pepsi.jukyo = "東京";

        pepsi.setGakunen(4);
        pepsi.setBunnya("情報");

        System.out.println("=== スマホセット前 ===");
        pepsi.shokai();

        // Sumaho インスタンスを生成して渡す（Sumaho クラスは既存のものを使用）
        Sumaho myphone = new Sumaho("Pixel 7", 6.3, 8);
        pepsi.setSumaho(myphone);

        System.out.println("=== スマホセット後 ===");
        pepsi.shokai();

        // スマホを手放す（null をセット）して再確認
        pepsi.setSumaho(null);
        System.out.println("=== スマホを手放した後 ===");
        pepsi.shokai();
    }
}

4.
staticフィールドは「クラス変数」とも呼ばれる。
インスタンスごとに固有の値を持つ通常のフィールドと異なり、クラスに1つだけ存在し、すべてのインスタンスで共有される。

~用途~
「全インスタンスで共通の情報」を保持する場合（例：生成されたオブジェクトの個数をカウントするカウンタ）
定数を定義する場合（public static final で宣言して定数として利用）
状態や設定を全体で共有したい場合
