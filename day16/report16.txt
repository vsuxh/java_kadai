import java.io.*;

public class MyCat {
    public static void main(String[] args) {
        if (args.length == 0) {
            System.out.println("ファイル名を指定してください");
            return;
        }

        try {
            FileReader fr = new FileReader(args[0]);
            BufferedReader br = new BufferedReader(fr);

            String buf;
            while ((buf = br.readLine()) != null) {
                System.out.println(buf);
            }

            br.close();
            fr.close();
        } catch (IOException ex) {
            System.out.println("ファイル読込みエラー");
            System.exit(1);
        }
    }
}

import java.io.*;

public class MyCatLine {
    public static void main(String[] args) {
        if (args.length == 0) {
            System.out.println("ファイル名を指定してください");
            return;
        }

        try {
            FileReader fr = new FileReader(args[0]);
            BufferedReader br = new BufferedReader(fr);

            String buf;
            String all = "";

            while ((buf = br.readLine()) != null) {
                all += buf;
                all += System.getProperty("line.separator");
            }

            br.close();
            fr.close();

            System.out.println(all);

        } catch (IOException ex) {
            System.out.println("ファイル読込みエラー");
            System.exit(1);
        }
    }
}

import java.io.*;

public class MyCopy {
    public static void main(String[] args) {
        if (args.length < 2) {
            System.out.println("コピー元とコピー先のファイル名を指定してください");
            return;
        }

        try {
            FileReader fr = new FileReader(args[0]);
            BufferedReader br = new BufferedReader(fr);

            FileWriter fw = new FileWriter(args[1]);
            BufferedWriter bw = new BufferedWriter(fw);

            String buf;
            while ((buf = br.readLine()) != null) {
                bw.write(buf);
                bw.newLine(); // 改行もコピー
            }

            br.close();
            fr.close();
            bw.close();
            fw.close();

        } catch (IOException ex) {
            System.out.println("ファイルコピーエラー");
            System.exit(1);
        }
    }
}
