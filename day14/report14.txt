import javax.swing.*;
import java.awt.*;

public class FlowTest extends JFrame{
	private JPanel panel0;
	private JButton button0, button1, button2;
	public FlowTest(){
		super();
	        this.button0 = new JButton();
		this.button1 = new JButton();
		this.button2 = new JButton();
		JPanel panel0 = new JPanel(); 
		panel0.setLayout(new FlowLayout());
		panel0.add(this.button0);
		panel0.add(this.button1);
		panel0.add(this.button2);
		super.getContentPane().add(panel0);
	}

	public static void main(String[] args){
		FlowTest a = new FlowTest();
		a.setSize(300, 300);
		a.setLocation(0,0);
		a.setVisible(true);
	}

}

public class CardTest extends JFrame{
        private JButton button;
        private JTextField textbox;
        private JLabel label;

        public CardTest(){
                super();
                this.button = new JButton("example");
                this.label = new JLabel("card");
                this.textbox = new JTextField("test", 10);

                JPanel  panel = new JPanel();
                panel.setLayout(new CardLayout());

                panel.add(this.button);
                panel.add(this.label);
                panel.add(this.textbox);

                getContentPane().add(panel, BorderLayout.CENTER);


        }

        public static void main(String[] args){
                CardTest a = new CardTest();
                a.setSize(600, 400);
                a.setLocation(0,0);
                a.setVisible(true);
        }
}
